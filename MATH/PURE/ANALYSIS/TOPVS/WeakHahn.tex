\begin{flalign*}
&\THM{WeakHahnBanach} :: \forall E : \TYPE{Banach} \. \forall x \in E : x \neq 0 \. \exists \lambda \in \mathcal{M}_{\mathsf{TVS(\mathbb{R})}}(E,\mathbb{R}) : \lambda(x) \neq 0 \\
& \LOGIC{Proof:} \\ 
&  \LOGIC{A} \; E : \TYPE{Banach}, \\
& \LOGIC{A} \; x \in E : x \neq 0, \\
& e \de x/\Vert x \Vert,  \\
& \lambda  \de \Lambda \alpha e \in \FUNC{span}(e) \. \alpha, \\
& \LOGIC{A} \; K : \TYPE{Subspace}(E) : \dim K = 2 : e \in K , \\
&  \de K \Rightarrow \exists v \in K : v \not \in \FUNC{span}(e) \; \LOGIC{E} \;, \\
&  e' \de v/\Vert v \Vert, \\   
&  \LOGIC{Extend}(\lambda, \Lambda \alpha e + \beta e' \in K \. \alpha)  ; \\
& \LOGIC{Choice} \Rightarrow \lambda : \mathcal{L}(E,\mathbb{R}) 
\end{flalign*}
