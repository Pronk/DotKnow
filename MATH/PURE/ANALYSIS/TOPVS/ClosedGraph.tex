& \LOGIC{A} \; T : \mathcal{M}_{\mathsf{TVS}(\mathbb{R})}:\TYPE{Iso}(\mathsf{SET})(E,F), \\

\LOGIC{E}, \\
& T^{-1}(v + w) = T^{-1}(Ta + Tb) = T^{-1}T(a + b ) = a + b = T^{-1}Ta + T^{-1}Tb = T^{-1}v + T^{-1}w \; \LOGIC{as} \; (1);  \\
& \LOGIC{A} \; v \in F, \\
& \LOGIC{A} \; r \in \mathbb{R}, \\
& \TYPE{Inverse}(\mathsf{SET})(T)(T^{-1})  \Rightarrow  \exists a \in E : Ta = v \;   \LOGIC{E}, \\
& T^{-1}rv = T^{-1}rTa = T^{-1}Tra = ra = rT^{-1}Ta = rT^{-1}v \; \LOGIC{as} \; (2);; \\
& \TYPE{Linear}(\mathbb{R})(T^{-1},(1),(2)) \Rightarrow T^{-1} :  \mathcal{M}_{\mathsf{NVS}(\mathbb{R})}(F,E)
\; \LOGIC{as} \; (3), \\ 
& \LOGIC{A} \;  y: \TYPE{Convergent}(F), \\
&  Y \de \lim_{n \to \infty} y_n, X \de T^{-1}Y, x \de T^{-1}y, \\
&  0 = \lim_{n \to \infty} \Vert Y - y_n \Vert = \lim_{n \to \infty} \Vert TX - Tx_n \Vert =
   \lim_{n \to \infty} \Vert T(X - x_n) \Vert =
