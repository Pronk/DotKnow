	{
		\lambda^1 = \lambda
	}
	\Explain{ $\lambda^1(a,b) = b - a = \lambda(a,b)$ }
	\EndProof
	\\
	\Theorem{HypervolumeSubbaditivity}
	{
		\NewLine :		
		\forall d \in \Nat \.
		\forall (A,B) : \OI(\Reals^d) \.
		\forall I : \Countable \.
		\forall (a,b) : I \to \OI(\Reals^d) \. \NewLine \.
		\forall \aleph : (A,B) \subset \bigcup_{i \in I} (a_n,_n) \.
		\lambda^d(A,B) \le \sum_{i \in I} \lambda^d(A,B)(a_i,b_i)
